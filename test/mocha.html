PitchClassSet

constructor - from interval succession
let pitchClassSet = new PitchClassSet({
  from: "intervalSuccession",
  array: [1, 2, 3, 4]
});
assert.deepEqual(pitchClassSet.pcSet, [0, 1, 3, 6, 10]);
constructor - from interval succession, pitches transposed
let pitchClassSet = new PitchClassSet({
  from: 'intervalSuccession',
  array: [4, 4, 4, 4]
});
assert.deepEqual(pitchClassSet.pcSet, [0, 4, 8, 0, 4]);
constructor - from pitches
let pitchClassSet = new PitchClassSet({
  from: "pitches",
  array: [4, 6, 3, 4]
});
assert.deepEqual(pitchClassSet.pcSet, [4, 6, 3, 4]);
constructor - from pitches, pitches transposed
let pitchClassSet = new PitchClassSet({
  from: 'pitches',
  array: [60, 62, 64, 65]
});
assert.deepEqual(pitchClassSet.pcSet, [0, 2, 4, 5]);
